# mcts-reasoning v0.6 Refactor Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Refactor mcts-reasoning from a monolithic 7,585-line library into a composable ~3,500-line MCTS framework with pluggable ABCs, a benchmarking framework, MCP server, and CLI.

**Architecture:** Strip ~4,000 lines of speculative code, restructure around ABCs for every strategic decision point (provider, prompt, generator, evaluator, terminal, sampling, consensus). Stateless MCTS returns SearchState. Provider-per-file with detect(). Benchmarking framework enables empirical testing. MCP server is primary interface, CLI has feature parity.

**Tech Stack:** Python 3.10+, pyproject.toml, zero-dependency core, optional extras for providers (requests, openai, anthropic) and server (mcp)

**Spec:** `docs/superpowers/specs/2026-03-16-v06-architecture-design.md`

```python
# mcts_reasoning/types.py
"""Core data types for mcts-reasoning. Zero external dependencies."""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import TYPE_CHECKING, NewType, TypedDict

if TYPE_CHECKING:
    from .node import Node


class Message(TypedDict):
    role: str
    content: str


State = NewType("State", str)


def extend_state(state: State, continuation: str) -> State:
    """The single place where states grow by concatenation."""
    return State(f"{state}\n\n{continuation}")


@dataclass
class Continuation:
    text: State
    is_terminal: bool = False
    answer: str | None = None


@dataclass
class Evaluation:
    score: float
    explanation: str = ""


@dataclass
class TerminalCheck:
    is_terminal: bool
    answer: str | None = None


@dataclass
class SampledPath:
    nodes: list  # list of Node (avoid circular import with TYPE_CHECKING)
    answer: str | None = None
    value: float = 0.0
    visits: int = 0


@dataclass
class ConsensusResult:
    answer: str
    confidence: float
    distribution: dict[str, int] = field(default_factory=dict)
    paths_used: int = 0


@dataclass
class SearchState:
    root: "Node"
    question: str
    terminal_states: list[dict] = field(default_factory=list)
    simulations_run: int = 0
    exploration_constant: float = 1.414
    max_children_per_node: int = 3
    max_rollout_depth: int = 5

    def save(self, path: str) -> None:
        """Serialize to JSON."""
        import json
        from pathlib import Path as FilePath

        data = {
            "question": self.question,
            "simulations_run": self.simulations_run,
            "exploration_constant": self.exploration_constant,
            "max_children_per_node": self.max_children_per_node,
            "max_rollout_depth": self.max_rollout_depth,
            "terminal_states": self.terminal_states,
            "root": self.root.to_dict(),
        }
        FilePath(path).write_text(json.dumps(data, indent=2))

    @classmethod
    def load(cls, path: str) -> "SearchState":
        """Deserialize from JSON."""
        import json
        from pathlib import Path as FilePath
        from .node import Node

        data = json.loads(FilePath(path).read_text())
        return cls(
            root=Node.from_dict(data["root"]),
            question=data["question"],
            simulations_run=data["simulations_run"],
            terminal_states=data.get("terminal_states", []),
            exploration_constant=data.get("exploration_constant", 1.414),
            max_children_per_node=data.get("max_children_per_node", 3),
            max_rollout_depth=data.get("max_rollout_depth", 5),
        )
```

```python
"""Test utilities. Import explicitly: from mcts_reasoning.testing import MockLLMProvider"""
from __future__ import annotations
from ..types import Message, State, Continuation, Evaluation


class MockLLMProvider:
    """Mock LLM provider for testing. Returns responses sequentially."""
    def __init__(self, responses: list[str] | None = None):
        self.responses = responses or ["Mock response"]
        self.call_count = 0
        self.calls: list[list[Message]] = []

    def generate(self, messages: list[Message], max_tokens: int = 1000,
                 temperature: float = 0.7) -> str:
        self.calls.append(messages)
        response = self.responses[self.call_count % len(self.responses)]
        self.call_count += 1
        return response

    def is_available(self) -> bool:
        return True

    @classmethod
    def detect(cls, base_url: str | None = None) -> bool:
        return False

    def get_name(self) -> str:
        return "Mock"


class MockGenerator:
    """Mock generator for testing MCTS without an LLM."""
    def __init__(self, responses: list[str] | None = None, terminal_at: int = 3):
        self.responses = responses or ["Thinking step..."]
        self.call_count = 0
        self.terminal_at = terminal_at

    def generate(self, question: str, state: State, n: int = 1) -> list[Continuation]:
        from ..types import extend_state
        results = []
        for _ in range(n):
            text = self.responses[self.call_count % len(self.responses)]
            self.call_count += 1
            is_terminal = self.call_count >= self.terminal_at
            answer = "42" if is_terminal else None
            if is_terminal:
                text = f"{text} ANSWER: 42"
            new_state = extend_state(state, text)
            results.append(Continuation(text=new_state, is_terminal=is_terminal, answer=answer))
        return results


class MockEvaluator:
    """Mock evaluator for testing."""
    def __init__(self, score: float = 0.8):
        self.score = score

    def evaluate(self, question: str, state: State, answer: str) -> Evaluation:
        return Evaluation(score=self.score, explanation="mock")
```

- [ ] **Step 3: Rewrite mcts.py**

Key changes from v0.5:
1. `search()` creates a new root Node, runs simulations, returns `SearchState`. Does not store anything on `self`.
2. `continue_search(state, simulations)` takes an existing `SearchState`, runs more simulations on `state.root`, returns a new `SearchState` with updated stats.
3. **Inline the rollout**: no ActionSpace. The rollout loop calls `self.generator.generate(question, node.state, n=1)` directly. The Continuation already carries `is_terminal` and `answer`.
4. **Respect `max_children_per_node` during rollout**: check `len(node.children) < max_children_per_node` before adding a rollout child.
5. **`on_simulation` callback**: fire after each phase (select, expand, rollout, backprop).
6. **Iterative tree-walking** in all helper methods (find best leaf, etc.).
7. Remove all uses of `ActionSpace`, `ContinueAction`, `_continuation_info`.

The MCTS loop per simulation:
1. **Select**: UCB1 traversal from root to expandable leaf
2. **Expand**: call `generator.generate(question, node.state, n=1)`, add child from first continuation
3. **Rollout**: from expanded node, keep generating until terminal or max depth. Each step: `generator.generate()`, add child, check terminal.
4. **Backprop**: from terminal (or deepest node), walk up to root updating values.
